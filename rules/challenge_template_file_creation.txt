import pandas as pd
import numpy as np


def grade(submission: pd.DataFrame, answers: pd.DataFrame) -> float:
    '''Score numeric predictions using Root Mean Squared Error.
    TODO: Update column names to match your challenge.
    '''
    merged = answers.merge(submission, on='id', how='left')
    if merged['prediction'].isna().any():
        raise ValueError('Submission missing predictions for some rows')


    rmse = np.sqrt(((merged['target'] - merged['prediction']) ** 2).mean())
    return float(rmse)

Classification — accuracy:
import pandas as pd


def grade(submission: pd.DataFrame, answers: pd.DataFrame) -> float:
    '''Score classification predictions using accuracy.
    TODO: Update column names to match your challenge.
    '''
    merged = answers.merge(submission, on='id', how='left')
    if merged['prediction'].isna().any():
        raise ValueError('Submission missing predictions for some rows')


    accuracy = (merged['target'] == merged['prediction']).mean()
    return float(accuracy)


Template
Text generation — BLEU score:
import pandas as pd
from nltk.translate.bleu_score import sentence_bleu, SmoothingFunction


def grade(submission: pd.DataFrame, answers: pd.DataFrame) -> float:
    '''Score generated text using average BLEU score.
    TODO: Update column names to match your challenge.
    '''
    merged = answers.merge(submission, on='id', how='left')
    if merged['prediction'].isna().any():
        raise ValueError('Submission missing predictions for some rows')


    smooth = SmoothingFunction().method1
    scores = []
    for _, row in merged.iterrows():
        ref_tokens = row['target'].split()
        gen_tokens = row['prediction'].split()
        scores.append(sentence_bleu([ref_tokens], gen_tokens, smoothing_function=smooth))


    return float(sum(scores) / len(scores))

Exact match — string comparison:
import pandas as pd


def grade(submission: pd.DataFrame, answers: pd.DataFrame) -> float:
    '''Score predictions using exact string match (case-insensitive).
    TODO: Update column names to match your challenge.
    '''
    merged = answers.merge(submission, on='id', how='left')
    if merged['prediction'].isna().any():
        raise ValueError('Submission missing predictions for some rows')


    matches = merged['target'].str.strip().str.lower() == merged['prediction'].str.strip().str.lower()
    return float(matches.mean())

from pathlib import Path
import shutil


def prepare(raw: Path, public: Path, private: Path) -> None:
    '''Copy pre-split public/private directories from raw data.
    TODO: Verify the folder names match your dataset structure.
    '''
    raw_public = raw / 'public'
    raw_private = raw / 'private'


    if not raw_public.exists():
        raise FileNotFoundError(f"Expected 'public' directory in raw data at {raw_public}")
    if not raw_private.exists():
        raise FileNotFoundError(f"Expected 'private' directory in raw data at {raw_private}")


    for item in raw_public.iterdir():
        dest = public / item.name
        if item.is_dir():
            shutil.copytree(item, dest)
        else:
            shutil.copy2(item, dest)


    for item in raw_private.iterdir():
        dest = private / item.name
        if item.is_dir():
            shutil.copytree(item, dest)
        else:
            shutil.copy2(item, dest)

from pathlib import Path


def prepare(raw: Path, public: Path, private: Path) -> None:
    '''Split data.csv into train/test sets.
    TODO: Update file name, target column, and test_size for your data.
    '''
    import pandas as pd
    from sklearn.model_selection import train_test_split


    df = pd.read_csv(raw / 'data.csv')
    train, test = train_test_split(df, test_size=0.2, random_state=42)


    train.to_csv(public / 'train.csv', index=False)
    test.drop('target', axis=1).to_csv(public / 'test.csv', index=False)
    test[['id', 'target']].to_csv(private / 'answers.csv', index=False)
